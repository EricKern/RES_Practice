101  -- instruction ram selection
102 - function isel (a: std_logic_vector(mb 1 downto 0)) return boolean is
103  begin
104  if to_integer(unsigned(a(mb - 1 downto bb))) >= (bk - ni) then
105  return true;
106  else
107  return false;
108  end if;
109  end function;
110 
111  -- data ram selection
112 - function dsel (a: std_logic_vector(mb 1 downto 0)) return boolean is
113  begin
114  if
115  to_integer(unsigned(a(mb - 1 downto bb))) < (bk - ni) and
116  to_integer(unsigned(a(mb - 1 downto bb))) >= (bk - ni - nd) then
117  return true;
118  else
119  return false;
120  end if;
121  end function;
122 
123  -- ram bits
124  function mbits(b: integer) return integer is
125  begin
126  for i in 0 to bk - 1 loop
127  if 2**i = b then
128  return (i + bb);
129  elsif 2**i > b then
130  return (i + bb);
131  end if;
132  end loop;
133  end function;
134 
135  -- higher iram addresses: return exact number of bits
136 - function iaddr(a: std_logic_vector(mb 1 downto 0)) return std_logic_vector is
137  variable r: std_logic_vector(mb - bb - 1 downto 0) := a(mb - 1 downto bb);
138  variable b: integer;
139  begin
140  b := to_integer(unsigned(r));
141  for i in 0 to ni - 1 loop -- loop over ni blocks
142  if (i + bk - ni) = to_integer(unsigned(r)) then
143  b := i;
144  --return std_logic_vector(to_unsigned(i,mb - bb)) & a(bb - 1 downto 0);
145  end if;
146  end loop;
147  return std_logic_vector(to_unsigned(b,mbits(ni) - bb)) & a(bb - 1 downto 0);
148  end function;
149 
150  -- higher dram addresses: return exact number of bits
151 - function daddr(a: std_logic_vector(mb 1 downto 0)) return std_logic_vector is
152  variable r: std_logic_vector(mb - bb - 1 downto 0) := a(mb - 1 downto bb);
153  variable b: integer;
154  begin
155  b := to_integer(unsigned(r));
156  for i in 0 to nd - 1 loop -- loop over ni blocks
157  if (i + bk - ni - nd) = to_integer(unsigned(r)) then
158  b := i;
159  --return std_logic_vector(to_unsigned(i,mb - bb)) & a(bb - 1 downto 0);
160  end if;
161  end loop;
162  return std_logic_vector(to_unsigned(b,mbits(nd) - bb)) & a(bb - 1 downto 0);
163  end function;
